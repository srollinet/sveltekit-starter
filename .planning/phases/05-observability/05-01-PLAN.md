// See: https://svelte.dev/docs/kit/observability
import { NodeSDK } from '@opentelemetry/sdk-node';
import { getNodeAutoInstrumentations } from '@opentelemetry/auto-instrumentations-node';
import { OTLPTraceExporter } from '@opentelemetry/exporter-trace-otlp-proto';
import { createAddHookMessageChannel } from 'import-in-the-middle';
import { register } from 'node:module';

// Required for ESM module hooking — must run before NodeSDK instantiation
const { registerOptions } = createAddHookMessageChannel();
register('import-in-the-middle/hook.mjs', import.meta.url, registerOptions);

**3. Create `src/lib/server/logger.ts`** (per D-10, D-11, D-12, D-13):

```typescript
// src/lib/server/logger.ts
// Server-only — never import this in client-side code
import pino from 'pino';
import { trace, context } from '@opentelemetry/api';

export const logger = pino({
  level: process.env.LOG_LEVEL ?? 'info',
  mixin() {
    const span = trace.getSpan(context.active());
    if (!span) return {};
    const ctx = span.spanContext();
    return {
      trace_id: ctx.traceId,
      span_id: ctx.spanId,
      trace_flags: ctx.traceFlags,
    };
  },
  formatters: {
    level(label: string) {
      // Return label string instead of numeric level for readability
      return { level: label };
    },
  },
});
```

Note: logger.ts reads LOG_LEVEL from `process.env` directly (not from `$lib/server/env`) because pino initializes at module load time — before the Zod env validation runs. The Zod schema still validates LOG_LEVEL at startup; this is belt-and-suspenders.

**4. Update `src/hooks.server.ts`** (per D-21) — add separate otelHandle, wrap with sequence() for Phase 6 composability:

```typescript
// src/hooks.server.ts
import '$lib/server/env';
import type { Handle } from '@sveltejs/kit';
import { sequence } from '@sveltejs/kit/hooks';
import { trace, context } from '@opentelemetry/api';

// Separate handle function for OTEL span-to-locals enrichment.
// Kept as a named export so Phase 6 can compose it with sequence(noseconeHandle, otelHandle).
const otelHandle: Handle = async ({ event, resolve }) => {
  const span = trace.getSpan(context.active());
  if (span) {
    const ctx = span.spanContext();
    event.locals.traceId = ctx.traceId;
    event.locals.spanId = ctx.spanId;
  }
  return resolve(event);
};

// Use sequence() now so Phase 6 can add handles before/after without refactoring
export const handle = sequence(otelHandle);
```
